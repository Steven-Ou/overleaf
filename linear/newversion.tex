% SECTION 1: EIGENVALUES (5.1, 5.2)
% ============================================================

\begin{algobox}{ALGORITHM: Finding Eigenvalues ($\lambda$) & Vectors ($\mathbf{x}$)}
\begin{enumerate}
    \item \textbf{Find $\lambda$:} Solve $\det(A - \lambda I) = 0$.
    \begin{itemize}
        \item \textbf{3x3 Trick:} Before expanding, try row ops ($R_i + R_j$) to create zeros or factor out terms like $(2-\lambda)$.
    \end{itemize}
    \item \textbf{Find $\mathbf{x}$:} For each $\lambda$, solve $(A - \lambda I)\mathbf{x} = \mathbf{0}$.
    \begin{itemize}
        \item Plug $\lambda$ into matrix. Row reduce.
        \item \textbf{Must} have a free variable (row of zeros).
        \item Write solution in parametric vector form.
    \end{itemize}
\end{enumerate}
\end{algobox}

\begin{exbox}{EXAMPLE: Eigenvalues/Vectors (5.1, 5.2)}
Find eigen-stuff for $A = \begin{bmatrix} 4 & -2 \\ 1 & 1 \end{bmatrix}$.
\begin{enumerate}
    \item \textbf{Char Eq:} $\det(A - \lambda I) = \det \begin{bmatrix} 4-\lambda & -2 \\ 1 & 1-\lambda \end{bmatrix}$
    $$ = (4-\lambda)(1-\lambda) - (-2)(1) $$
    $$ = (\lambda^2 - 5\lambda + 4) + 2 = \lambda^2 - 5\lambda + 6 $$
    $$ (\lambda - 2)(\lambda - 3) = 0 \implies \lambda = 2, 3 $$
    \item \textbf{Eigenvector for $\lambda = 2$:}
    $$ (A - 2I) = \begin{bmatrix} 2 & -2 \\ 1 & -1 \end{bmatrix} \sim \begin{bmatrix} 1 & -1 \\ 0 & 0 \end{bmatrix} $$
    $$ x_1 - x_2 = 0 \implies x_1 = x_2. \quad \text{Free: } x_2. $$
    $$ \mathbf{x} = x_2 \begin{bmatrix} 1 \\ 1 \end{bmatrix} \implies \text{Basis: } \left\{ \begin{bmatrix} 1 \\ 1 \end{bmatrix} \right\} $$
\end{enumerate}
\end{exbox}

% SECTION 2: RANK & NULLITY (2.8, 2.9)
% ============================================================

\begin{algobox}{ALGORITHM: Rank, Nullity, Col(A), Nul(A)}
\begin{enumerate}
    \item \textbf{Row Reduce} $A$ to REF. Identify Pivot Columns.
    \item \textbf{Rank} = \# of Pivots.
    \item \textbf{Nullity} = \# of Free Variables (Total Cols - Rank).
    \item \textbf{Basis Col(A):} Pivot columns from \textbf{ORIGINAL} matrix.
    \item \textbf{Basis Nul(A):} Solve $A\mathbf{x}=\mathbf{0}$ (Parametric Form).
\end{enumerate}
\end{algobox}

\begin{exbox}{EXAMPLE: Quiz 7 Problem (2.8, 2.9)}
$A = \begin{bmatrix} 4 & 5 & 9 & -2 \\ 6 & 5 & 1 & 12 \\ 3 & 4 & 8 & -3 \end{bmatrix} \sim \begin{bmatrix} \mathbf{1} & 2 & 6 & -5 \\ 0 & \mathbf{1} & 5 & -6 \\ 0 & 0 & 0 & 0 \end{bmatrix}$
\begin{enumerate}
    \item \textbf{Pivots:} Columns 1 and 2.
    \item \textbf{Rank:} 2. \quad \textbf{Nullity:} $4 - 2 = 2$.
    \item \textbf{Basis Col(A):} Use Cols 1 \& 2 of \textbf{original} A.
    $$ \left\{ \begin{bmatrix} 4 \\ 6 \\ 3 \end{bmatrix}, \begin{bmatrix} 5 \\ 5 \\ 4 \end{bmatrix} \right\} $$
    \item \textbf{Basis Nul(A):} Solve $Ax=0$. $x_3, x_4$ are free.
    $$ x_1 + 2x_2 + 6x_3 - 5x_4 = 0 $$
    $$ x_2 + 5x_3 - 6x_4 = 0 \implies x_2 = -5x_3 + 6x_4 $$
    $$ x_1 = -2(-5x_3 + 6x_4) - 6x_3 + 5x_4 = 4x_3 - 7x_4 $$
    $$ \mathbf{x} = x_3 \begin{bmatrix} 4 \\ -5 \\ 1 \\ 0 \end{bmatrix} + x_4 \begin{bmatrix} -7 \\ 6 \\ 0 \\ 1 \end{bmatrix} $$
\end{enumerate}
\end{exbox}

% SECTION 3: DETERMINANTS (3.1, 3.2)
% ============================================================

\begin{algobox}{ALGORITHM: Determinant Strategy}
\begin{enumerate}
    \item \textbf{Row Ops:} $R_i + kR_j$ (No change). Swap (Negate).
    \item \textbf{Triangular:} Multiply diagonal.
    \item \textbf{Cofactor:} Use row with most zeros.
    \item \textbf{Properties:} $\det(AB) = \det A \det B$, $\det(A^{-1}) = \frac{1}{\det A}$, $\det(A^T) = \det A$.
    \item \textbf{Scaling:} $\det(kA) = k^n \det(A)$ (for $n \times n$).
\end{enumerate}
\end{algobox}

\begin{exbox}{EXAMPLE: Quiz 8 Problem (3.1, 3.2)}
$A = \begin{bmatrix} 2 & -2 & 3 \\ 3 & 1 & 2 \\ 1 & 3 & -1 \end{bmatrix}$. Find $\det(A)$.
\begin{itemize}
    \item \textbf{Cofactor expansion} (Row 1):
    $$ 2 \begin{vmatrix} 1 & 2 \\ 3 & -1 \end{vmatrix} - (-2) \begin{vmatrix} 3 & 2 \\ 1 & -1 \end{vmatrix} + 3 \begin{vmatrix} 3 & 1 \\ 1 & 3 \end{vmatrix} $$
    $$ = 2(-1-6) + 2(-3-2) + 3(9-1) $$
    $$ = 2(-7) + 2(-5) + 3(8) = -14 - 10 + 24 = 0 $$
    \item \textbf{Invertible?} No, because $\det(A) = 0$.
\end{itemize}
\end{exbox}

\begin{exbox}{EXAMPLE: Determinant Properties (Quiz 8)}
If $\det(A) = -2$ and $A$ is $3 \times 3$:
\begin{itemize}
    \item $\det(5A) = 5^3 \det(A) = 125(-2) = -250$.
    \item $\det(A^3) = (\det A)^3 = (-2)^3 = -8$.
\end{itemize}
\end{exbox}

% SECTION 4: INVERSE MATRIX (2.2, 2.3)
% ============================================================

\begin{algobox}{ALGORITHM: Finding Inverse $A^{-1}$}
\begin{enumerate}
    \item \textbf{2x2 Shortcut:} $\frac{1}{ad-bc} \begin{bmatrix} d & -b \\ -c & a \end{bmatrix}$
    \item \textbf{3x3+ Algorithm:} Augment $[A \mid I]$. Row reduce.
    \item If you get $[I \mid B]$, then $B = A^{-1}$.
    \item If row of zeros appears on left, not invertible.
\end{enumerate}
\end{algobox}

\begin{exbox}{EXAMPLE: 2x2 Inverse}
$A = \begin{bmatrix} 1 & 2 \\ 3 & 4 \end{bmatrix}$. $\det = 4-6 = -2$.
$$ A^{-1} = \frac{1}{-2} \begin{bmatrix} 4 & -2 \\ -3 & 1 \end{bmatrix} = \begin{bmatrix} -2 & 1 \\ 1.5 & -0.5 \end{bmatrix} $$
\end{exbox}

\begin{thmbox}{TEST: Existence \& Uniqueness (Thm 1.2)}
Row reduce augmented matrix $[A \mid \mathbf{b}]$.
\begin{enumerate}
    \item \textbf{Consistent (Existence):} No row like $[0 \dots 0 \mid \text{nonzero}]$.
    \item \textbf{Unique Solution:} Consistent AND No free variables (Pivot in every column of $A$).
    \item \textbf{Infinite Solutions:} Consistent AND At least one free variable.
\end{enumerate}
\end{thmbox}

\begin{thmbox}{TEST: Do cols of $A$ span $\mathbb{R}^m$? (Thm 1.4)}
\begin{itemize}
    \item \textbf{Yes $\iff$} $A$ has a pivot position in \textbf{every row}.
    \item \textbf{Check:} Row reduce. Do you have a row of zeros? If yes, they do NOT span $\mathbb{R}^m$.
\end{itemize}
\end{thmbox}

\begin{thmbox}{TEST: Solutions to $A\mathbf{x}=\mathbf{b}$ (Thm 1.6)}
Solution = Particular + Homogeneous ($ \mathbf{x} = \mathbf{p} + \mathbf{v}_h $).
\begin{itemize}
    \item $\mathbf{p}$: Any vector solving $A\mathbf{p} = \mathbf{b}$.
    \item $\mathbf{v}_h$: The general solution to $A\mathbf{x} = \mathbf{0}$.
\end{itemize}
\end{thmbox}

% SECTION 6: LINEAR INDEPENDENCE (1.7)
% ============================================================

\begin{thmbox}{TEST: Linear Independence (Thm 1.7)}
Set of vectors $\{\mathbf{v}_1, \dots, \mathbf{v}_p\}$. Form matrix $A$.
\begin{enumerate}
    \item \textbf{Pivot in every column?} $\to$ Independent.
    \item \textbf{Free variable exists?} $\to$ Dependent.
\end{enumerate}
\textbf{Instant Checks (Dependent if...):}
\begin{itemize}
    \item More vectors than rows ($p > n$).
    \item The set contains the $\mathbf{0}$ vector.
    \item Multiples: One vector is a scalar multiple of another.
\end{itemize}
\end{thmbox}

% ============================================================
% SECTION 7: LINEAR TRANSFORMATIONS (1.8, 1.9)
% ============================================================

\begin{algobox}{ALGORITHM: Finding Standard Matrix $A$}
For a transformation $T: \mathbb{R}^n \to \mathbb{R}^m$:
\begin{enumerate}
    \item Find $T(\mathbf{e}_1), T(\mathbf{e}_2), \dots$ (Transform the identity cols).
    \item $A = [T(\mathbf{e}_1) \quad T(\mathbf{e}_2) \quad \dots \quad T(\mathbf{e}_n)]$.
\end{enumerate}
\end{algobox}

\begin{thmbox}{TEST: One-to-One & Onto (Thm 1.11/1.12)}
Row reduce standard matrix $A$.
\begin{itemize}
    \item \textbf{Onto (Surjective):} Pivot in every \textbf{ROW}. (Cols span $\mathbb{R}^m$).
    \item \textbf{One-to-One (Injective):} Pivot in every \textbf{COLUMN}. (Cols are Linearly Independent).
\end{itemize}
\end{thmbox}
