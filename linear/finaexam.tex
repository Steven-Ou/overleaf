\section*{I. Systems and RREF (1.1, 1.2)}
\begin{thmbox}{Existence and Uniqueness (Thm 1.2)}
\begin{itemize}
    \item \textbf{Existence:} Consistent $\iff [A \mid \mathbf{b}]$ has \textbf{NO pivot} in the rightmost column.
    \item \textbf{Unique:} Consistent AND \textbf{No free variables} (Pivot in every column of $A$).
    \item \textbf{Infinite:} Consistent AND at least one free variable.
\end{itemize}
\end{thmbox}

\begin{exbox}{Example: Parametric Vector Form (PVF)}
If $x_1 = 5-3x_3$ and $x_2 = -3-2x_3$ with $x_3$ free, the solution is:
$\mathbf{x} = \begin{bmatrix} 5 \\ -3 \\ 0 \end{bmatrix} + x_3 \begin{bmatrix} -3 \\ -2 \\ 1 \end{bmatrix}$ where $\mathbf{p}$ is particular and $\mathbf{v}_h$ is homogeneous.
\end{exbox}

\section*{II. Vectors, Span, and Independence (1.3-1.7)}
\begin{thmbox}{Span and Independence Tests}
\begin{itemize}
    \item \textbf{Cols span $\mathbb{R}^m \iff$} $A$ has a \textbf{pivot in every row}.
    \item \textbf{LI $\iff$} $Ax=0$ has \textbf{only trivial solution} (Pivot in every column).
    \item \textbf{LD Instant Checks:} More vectors than entries ($p>n$); contains $\mathbf{0}$ vector; one vector is a multiple of another.
\end{itemize}
\end{thmbox}

\section*{III. Linear Transformations (1.8, 1.9)}
\begin{algobox}{Standard Matrix $A$ & Mapping Tests}
\begin{itemize}
    \item \textbf{Standard Matrix:} $A = [T(\mathbf{e}_1) \dots T(\mathbf{e}_n)]$.
    \item \textbf{Onto (Surjective):} Pivot in every \textbf{row}.
    \item \textbf{One-to-One (Injective):} Pivot in every \textbf{column}.
\end{itemize}
\end{algobox}

\section*{IV. Matrix Algebra & Inverses (2.1-2.3)}
\begin{algobox}{Inverse Properties & 2x2 Shortcut}
\begin{itemize}
    \item \textbf{Inverse Algorithm:} $[A \mid I] \sim [I \mid A^{-1}]$.
    \item \textbf{2x2 Shortcut:} $A^{-1} = \frac{1}{ad-bc} \begin{bmatrix} d & -b \\ -c & a \end{bmatrix}$.
    \item \textbf{Properties:} $(AB)^{-1} = B^{-1}A^{-1}$; $(A^T)^{-1} = (A^{-1})^T$.
\end{itemize}
\end{algobox}
\end{multicols}

\begin{multicols}{2}
\section*{V. Determinants (3.1, 3.2)}
\begin{thmbox}{Properties & Strategy}
\begin{itemize}
    \item \textbf{Scaling:} $\det(kA) = k^n \det(A)$.
    \item \textbf{Row Ops:} Swap = $-det$; Replacement = No change; Scaling $R_i$ by $k$ = $k \cdot det$.
    \item \textbf{Inverse:} $\det(A^{-1}) = 1/\det(A)$.
    \item \textbf{Invertible $\iff \det(A) \neq 0$}.
\end{itemize}
\end{thmbox}

\section*{VI. Subspaces, Rank & Basis (2.8, 2.9)}
\begin{algobox}{Basis & Dimension Rules}
\begin{itemize}
    \item \textbf{Basis Col(A):} Pivot columns of \textbf{ORIGINAL} matrix.
    \item \textbf{Basis Nul(A):} Solve $Ax=0$ and write in parametric form.
    \item \textbf{Rank-Nullity:} $rank(A) + nullity(A) = n$ (total cols).
\end{itemize}
\end{algobox}

\section*{VII. Vlamis Style Logic & Traps}
\begin{exbox}{Proofs & Counter-examples}
\begin{itemize}
    \item \textbf{Subspace Trap:} $W = \{[x,y]^T : xy \geq 0\}$ is NOT a subspace (fails addition closure).
    \item \textbf{Dependency Arg:} If $AB$ has a zero column and $B$ does not, $A$ must have dependent columns.
    \item \textbf{Identical Cols:} If square $A$ has $\vec{a}_i = \vec{a}_j$, then $A$ is not invertible.
\end{itemize}
\end{exbox}

\section*{VIII. THE INVERSE MATRIX THEOREM (IMT)}
\begin{tcolorbox}[colback=myyellow!10!white, colframe=Orange, title=CUMULATIVE EQUIVALENTS]
For square $n \times n$ matrix $A$, these are equivalent: \\
1. $A$ is invertible $\iff \det(A) \neq 0$. \\
2. $A \sim I_n$ (has $n$ pivots). \\
3. $Ax=0$ has only the trivial solution. \\
4. Columns are LI and span $\mathbb{R}^n$. \\
5. $rank(A) = n$ and $nullity(A) = 0$. \\
6. $0$ is NOT an eigenvalue.
\end{tcolorbox}
\end{multicols}

\section*{IX. Eigenvalues & Eigenvectors (5.1, 5.2)}
\begin{algobox}{Algorithm: Finding $\lambda$ and $\mathbf{x}$}
1. \textbf{Char Eq:} Solve $\det(A - \lambda I) = 0$. \\
2. \textbf{Find $\mathbf{x}$:} For each $\lambda$, solve $(A - \lambda I)\mathbf{x} = \mathbf{0}$. \\
3. \textbf{Note:} Eigenspace is $Nul(A - \lambda I)$.
\end{algobox}

\begin{exbox}{Example: Diagonalization}
$A = PDP^{-1}$ if $A$ has $n$ LI eigenvectors. $D$ is diagonal matrix of eigenvalues, $P$ is matrix of eigenvectors.
\end{exbox}

\section*{X. Markov Chains (App 10.1)}
\begin{algobox}{Steady State Vector $\mathbf{w}$}
\begin{itemize}
    \item \textbf{Matrix:} Stochastic matrix $M$ has columns summing to 1.
    \item \textbf{Find $\mathbf{w}$:} Solve $(M - I)\mathbf{x} = \mathbf{0}$.
    \item \textbf{Normalization:} Scale the result so the sum of entries is 1.
\end{itemize}
\end{algobox}

\begin{exbox}{Steady State Example}
$M = \begin{bmatrix} 0.7 & 0.2 \\ 0.3 & 0.8 \end{bmatrix}$. Solve $\begin{bmatrix} -0.3 & 0.2 \\ 0.3 & -0.2 \end{bmatrix}x=0$. \\
$3x_1 = 2x_2 \to$ basis $\begin{bmatrix} 2 \\ 3 \end{bmatrix}$. Normalized: $\mathbf{w} = \begin{bmatrix} 0.4 \\ 0.6 \end{bmatrix}$.
\end{exbox}
\end{multicols}

\begin{multicols}{2}
\section*{XI. Orthogonality & Projections (6.1-6.3)}
\begin{thmbox}{Core Theorems}
\begin{itemize}
    \item \textbf{Complement:} $v \in W \cap W^\perp \Rightarrow v = \vec{0}$.
    \item \textbf{Projection:} $\hat{y} = proj_W y = \sum \frac{y \cdot u_i}{u_i \cdot u_i} u_i$.
    \item \textbf{Orthogonal Matrix:} $Q^T Q = I$; preserves norms.
\end{itemize}
\end{thmbox}

\section*{XII. Gram-Schmidt & QR (6.4)}
\begin{algobox}{Algorithms}
\begin{itemize}
    \item \textbf{Gram-Schmidt:} $v_1 = x_1$; $v_2 = x_2 - proj_{v_1} x_2$.
    \item \textbf{QR Decomposition:} $A = QR$ where $Q$ is orthogonal and $R$ is upper triangular.
\end{itemize}
\end{algobox}

\section*{XIII. Least Squares (6.5, 6.6)}
\begin{algobox}{Normal Equations}
\begin{itemize}
    \item \textbf{Goal:} Minimize $\|b - Ax\|$.
    \item \textbf{Normal Equations:} Solve $A^T A \hat{x} = A^T b$.
    \item \textbf{Uniqueness:} Unique if columns of $A$ are LI.
\end{itemize}
\end{algobox}
\end{multicols}
